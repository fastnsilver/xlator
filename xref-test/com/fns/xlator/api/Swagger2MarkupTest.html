20  import io.github.robwin.markup.builder.MarkupLanguage;
21  import springfox.documentation.staticdocs.Swagger2MarkupResultHandler;
22  
23  @WebAppConfiguration
24  @RunWith(SpringJUnit4ClassRunner.class)
25  @SpringApplicationConfiguration(classes = App.class)
26  public class Swagger2MarkupTest {
27  
28      @Autowired
29      private WebApplicationContext context;
30  
31      private MockMvc mockMvc;
32  
33      @Before
34      public void setUp() {
35          this.mockMvc = MockMvcBuilders.webAppContextSetup(this.context).build();
36      }
37  
38      @Test
39      public void convertSwaggerToAsciiDoc() throws Exception {
40          this.mockMvc.perform(get("/v2/api-docs")
41                  .accept(MediaType.APPLICATION_JSON))
42                  .andDo(Swagger2MarkupResultHandler
43                      .outputDirectory("target/docs/asciidoc/generated").build())
44                  .andExpect(status().isOk());
45      }
46  
47      @Test
48      public void convertSwaggerToMarkdown() throws Exception {
49          this.mockMvc.perform(get("/v2/api-docs")
50                  .accept(MediaType.APPLICATION_JSON))
51                  .andDo(Swagger2MarkupResultHandler.outputDirectory("target/docs/markdown/generated")
52                      .withMarkupLanguage(MarkupLanguage.MARKDOWN).build())
53                  .andExpect(status().isOk());
54      }
55  }
