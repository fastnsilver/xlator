001package com.fns.xlator.client.impl;
002
003import com.fns.xlator.client.api.TranslationService;
004import com.fns.xlator.model.Translation;
005
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008import org.springframework.beans.factory.annotation.Autowired;
009import org.springframework.boot.actuate.metrics.CounterService;
010import org.springframework.cache.annotation.Cacheable;
011import org.springframework.stereotype.Service;
012import org.springframework.util.Assert;
013import org.springframework.web.client.RestClientException;
014import org.springframework.web.client.RestTemplate;
015import org.springframework.web.util.UriComponentsBuilder;
016
017import java.net.URI;
018import java.util.Arrays;
019import java.util.Locale;
020import java.util.MissingResourceException;
021
022@Service
023public class FrenglyTranslationService implements TranslationService {
024
025    private final Logger log = LoggerFactory.getLogger(getClass());
026
027    private RestTemplate restTemplate;
028    private FrenglyClientSettings settings;
029    private CounterService counterService;
030
031    @Autowired
032    public FrenglyTranslationService(RestTemplate restTemplate, FrenglyClientSettings settings,
033            CounterService counterService) {
034        this.restTemplate = restTemplate;
035        this.settings = settings;
036        this.counterService = counterService;
037    }
038
039    private Locale parseLocale(String locale) {
040        String[] parts = locale.split("_");
041        switch (parts.length) {
042            case 3:
043                return new Locale(parts[0], parts[1], parts[2]);
044            case 2:
045                return new Locale(parts[0], parts[1]);
046            case 1:
047                return new Locale(parts[0]);
048            default:
049                throw new IllegalArgumentException("Invalid locale: " + locale);
050        }
051    }
052
053    private boolean isValidLocale(String code) {
054        try {
055            Locale locale = parseLocale(code);
056            return Arrays.asList(Locale.getAvailableLocales()).contains(locale);
057        } catch (MissingResourceException e) {
058            return false;
059        }
060    }
061
062    protected void validateTranslationArguments(String source, String target, String text) {
063        Assert.hasText(target, "[Assertion failed] - Target locale code must not be null, empty or blank!");
064        Assert.hasText(text, "[Assertion failed] - Text to translate must not be null, empty or blank!");
065        Assert.isTrue(isValidLocale(source), String.format("[Assertion failed] - Source [%s] is not a valid Locale!", source));
066        Assert.isTrue(isValidLocale(target), String.format("[Assertion failed] - Target [%s] is not a valid Locale!", target));
067    }
068
069    protected FrenglyTranslation tryRequest(String source, String target, String text, int retries) {
070        URI uri = UriComponentsBuilder
071                .newInstance()
072                .scheme("http")
073                .host(settings.getHost())
074                .port(80)
075                .queryParam("src", source)
076                .queryParam("dest", target)
077                .queryParam("text", text)
078                .queryParam("email", settings.getEmail())
079                .queryParam("password", settings.getPassword())
080                .queryParam("outformat", "json")
081                .build()
082                .encode()
083                .toUri();
084        log.info(String.format("Obtaining translation from %s", uri.toASCIIString()));
085
086        FrenglyTranslation result = null;
087        for (int i = 0; i < retries; i++) {
088            try {
089                result = restTemplate.getForObject(uri, FrenglyTranslation.class);
090                break;
091            } catch (RestClientException e) {
092                try {
093                    Thread.sleep((i + 1) * 500);
094                } catch (InterruptedException ie) {
095                    // do nothing
096                }
097            }
098            if (i == retries - 1) {
099                result = new FrenglyTranslation(source, target, text, "Could not obtain translation!"); // FIXME i18n response
100            }
101        }
102        return result;
103    }
104
105    @Override
106    @Cacheable(cacheNames = "translations")
107    public Translation obtainTranslation(String src, String target, String text) {
108        counterService.increment("services.frengly.translation.invoked");
109        String source = src;
110        if (source == null) {
111            source = settings.getDefaultSource();
112        }
113        validateTranslationArguments(source, target, text);
114        if (src.equals(target)) {
115            return new Translation(src, target, text, text);
116        }
117        FrenglyTranslation ft = tryRequest(source, target, text, settings.getRetries());
118        return new Translation(ft.getSrc(), ft.getDest(), ft.getText(), ft.getTranslation());
119    }
120
121}
