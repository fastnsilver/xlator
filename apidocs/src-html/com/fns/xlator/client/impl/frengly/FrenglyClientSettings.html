001package com.fns.xlator.client.impl.frengly;
002
003import org.hibernate.validator.constraints.NotEmpty;
004import org.springframework.beans.factory.annotation.Value;
005import org.springframework.boot.context.properties.ConfigurationProperties;
006
007@ConfigurationProperties(prefix = "app.frengly")
008public class FrenglyClientSettings {
009
010    private String host = "frengly.com";
011
012    @NotEmpty
013    private String email = "change@me.com";
014
015    @NotEmpty
016    private String password = "changeme";
017
018    private int retries = 10;
019
020    @Value("${app.defaults.locale}")
021    private String defaultSource;
022
023    public String getHost() {
024        return host;
025    }
026
027    public void setHost(String host) {
028        this.host = host;
029    }
030
031    public String getEmail() {
032        return email;
033    }
034
035    public void setEmail(String email) {
036        this.email = email;
037    }
038
039    public String getPassword() {
040        return password;
041    }
042
043    public void setPassword(String password) {
044        this.password = password;
045    }
046
047    public String getDefaultSource() {
048        return defaultSource;
049    }
050
051    public int getRetries() {
052        return retries;
053    }
054}
