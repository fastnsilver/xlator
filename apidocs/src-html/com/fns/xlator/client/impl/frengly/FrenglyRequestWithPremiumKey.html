020package com.fns.xlator.client.impl.frengly;
021
022import com.fasterxml.jackson.annotation.JsonCreator;
023import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
024import com.fasterxml.jackson.annotation.JsonInclude;
025import com.fasterxml.jackson.annotation.JsonInclude.Include;
026import com.fasterxml.jackson.annotation.JsonProperty;
027import com.fasterxml.jackson.annotation.JsonPropertyOrder;
028
029@JsonInclude(content = Include.NON_NULL)
030@JsonIgnoreProperties(ignoreUnknown = true)
031@JsonPropertyOrder(value = { "src", "dest", "text", "email", "password", "premiumKey" })
032public class FrenglyRequestWithPremiumKey {
033
034    private String src;
035    private String dest;
036    private String text;
037    private String email;
038    private String password;
039    private String premiumKey;
040
041    @JsonCreator
042    public FrenglyRequestWithPremiumKey(
043                @JsonProperty("src") String src, 
044                @JsonProperty("dest") String dest,
045            @JsonProperty("text") String text, 
046            @JsonProperty("email") String email,
047            @JsonProperty("password") String password,
048            @JsonProperty("premiumKey") String premiumKey) {
049        this.src = src;
050        this.dest = dest;
051        this.text = text;
052        this.email = email;
053        this.password = password;
054        this.premiumKey = premiumKey;
055    }
056
057
058    public String getSrc() {
059        return src;
060    }
061
062    public String getDest() {
063        return dest;
064    }
065
066    public String getText() {
067        return text;
068    }
069
070    public String getEmail() {
071        return email;
072    }
073    
074    public String getPassword() {
075        return password;
076    }
077
078    public String getPremiumKey() {
079        return premiumKey;
080    }
081}
