001package com.fns.xlator.client.impl;
002
003import org.springframework.util.Assert;
004
005import java.util.Arrays;
006import java.util.Locale;
007import java.util.MissingResourceException;
008
009public class ValidationUtils {
010
011    // FIXME i18n responses
012
013    private static Locale parseLocale(String locale) {
014        String[] parts = locale.split("_");
015        switch (parts.length) {
016            case 3:
017                return new Locale(parts[0], parts[1], parts[2]);
018            case 2:
019                return new Locale(parts[0], parts[1]);
020            case 1:
021                return new Locale(parts[0]);
022            default:
023                throw new IllegalArgumentException("Invalid locale: " + locale);
024        }
025    }
026
027    private static boolean isValidLocale(String code) {
028        try {
029            Locale locale = parseLocale(code);
030            return Arrays.asList(Locale.getAvailableLocales()).contains(locale);
031        } catch (MissingResourceException e) {
032            return false;
033        }
034    }
035
036    public static void validateTranslationArguments(String source, String target, String text) {
037        Assert.hasText(target, "[Assertion failed] - Target locale code must not be null, empty or blank!");
038        Assert.hasText(text, "[Assertion failed] - Text to translate must not be null, empty or blank!");
039        Assert.isTrue(isValidLocale(source), String.format("[Assertion failed] - Source [%s] is not a valid Locale!", source));
040        Assert.isTrue(isValidLocale(target), String.format("[Assertion failed] - Target [%s] is not a valid Locale!", target));
041    }
042
043}
