020package com.fns.xlator.client.impl.google;
021
022import java.net.URI;
023import java.util.concurrent.TimeUnit;
024
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027import org.springframework.beans.factory.annotation.Autowired;
028import org.springframework.cache.annotation.Cacheable;
029import org.springframework.http.HttpStatus;
030import org.springframework.web.client.HttpClientErrorException;
031import org.springframework.web.client.RestTemplate;
032import org.springframework.web.util.UriComponentsBuilder;
033
034import com.fns.xlator.client.api.TranslationService;
035import com.fns.xlator.client.config.AppConfig;
036import com.fns.xlator.client.impl.TranslationException;
037import com.fns.xlator.client.impl.ValidationUtils;
038import com.fns.xlator.model.Translation;
039
040public class GoogleTranslationService implements TranslationService {
041
042    private final Logger log = LoggerFactory.getLogger(getClass());
043
044    private RestTemplate restTemplate;
045    private GoogleClientSettings settings;
046
047    @Autowired
048    public GoogleTranslationService(RestTemplate restTemplate, GoogleClientSettings settings) {
049        this.restTemplate = restTemplate;
050        this.settings = settings;
051    }
052
053    // @see https://cloud.google.com/translate/v2/using_rest#Translate
054    protected GoogleTranslation tryRequest(String source, String target, String text, int retries) {
055        URI uri = UriComponentsBuilder
056                .newInstance()
057                .scheme("https")
058                .host(settings.getHost())
059                .path("language/translate/v2")
060                .queryParam("key", settings.getKey())
061                .queryParam("source", source)
062                .queryParam("target", target)
063                .queryParam("q", text)
064                .build()
065                .encode().toUri();
066        log.info(String.format("Obtaining translation from %s", uri.toASCIIString()));
067
068        GoogleTranslation result = null;
069        for (int i = 0; i < retries; i++) {
070            try {
071                result = restTemplate.getForObject(uri, GoogleTranslation.class);
072                break;
073            } catch (final HttpClientErrorException e) {
074                if (e.getStatusCode() == HttpStatus.NOT_FOUND || e.getStatusCode() == HttpStatus.FORBIDDEN) {
075                    throw new TranslationException(source, target, text, e.getMessage());
076                }
077                try {
078                    // 0.5 ms multiplier delay for subsequent retries
079                    TimeUnit.MICROSECONDS.sleep((i + 1) * 500);
080                } catch (final InterruptedException ie) {
081                    // do nothing
082                }
083            }
084        }
085        return result;
086    }
087
088    @Override
089    @Cacheable(cacheNames = AppConfig.CACHE_NAME)
090    public Translation obtainTranslation(String src, String target, String text) {
091        String source = src;
092        if (source == null) {
093            source = settings.getDefaultSource();
094        }
095        ValidationUtils.validateTranslationArguments(source, target, text);
096        if (src.equals(target)) {
097            return new Translation(src, target, text, text);
098        }
099        GoogleTranslation gt = tryRequest(source, target, text, settings.getRetries());
100        return new Translation(source, target, text, gt.getData().getTranslations().get(0).getTranslatedText());
101    }
102
103}
