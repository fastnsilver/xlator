001package com.fns.xlator.client.impl.google;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import com.fasterxml.jackson.annotation.JsonPropertyOrder;
007
008@JsonIgnoreProperties(ignoreUnknown = true)
009@JsonPropertyOrder({ "data" })
010public class GoogleTranslation {
011
012    private Data data;
013
014    @JsonCreator
015    public GoogleTranslation(@JsonProperty("data") Data data) {
016        this.data = data;
017    }
018
019    public GoogleTranslation(String errorText) {
020        this.data = Data.error(errorText);
021    }
022
023    public Data getData() {
024        return data;
025    }
026}
