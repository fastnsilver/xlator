001package com.fns.xlator.client.impl.google;
002
003import org.hibernate.validator.constraints.NotEmpty;
004import org.springframework.beans.factory.annotation.Value;
005import org.springframework.boot.context.properties.ConfigurationProperties;
006
007@ConfigurationProperties(prefix = "app.google")
008public class GoogleClientSettings {
009
010    private String host = "www.googleapis.com";
011
012    @NotEmpty
013    private String key = "changeme";
014
015    private int retries = 10;
016
017    @Value("${app.defaults.locale}")
018    private String defaultSource;
019
020    public String getHost() {
021        return host;
022    }
023
024    public void setHost(String host) {
025        this.host = host;
026    }
027
028    public String getKey() {
029        return key;
030    }
031
032    public void setKey(String key) {
033        this.key = key;
034    }
035
036    public String getDefaultSource() {
037        return defaultSource;
038    }
039
040    public int getRetries() {
041        return retries;
042    }
043}
