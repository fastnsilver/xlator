001package com.fns.xlator.client.impl.google;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonProperty;
005import com.fasterxml.jackson.annotation.JsonPropertyOrder;
006
007@JsonPropertyOrder({ "translatedText" })
008public class Translation {
009
010    private String translatedText;
011
012    @JsonCreator
013    public Translation(@JsonProperty("translatedText") String translatedText) {
014        this.translatedText = translatedText;
015    }
016
017    public String getTranslatedText() {
018        return translatedText;
019    }
020}
