001package com.fns.xlator.client.api;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005import org.springframework.beans.factory.annotation.Autowired;
006import org.springframework.boot.actuate.metrics.CounterService;
007import org.springframework.cache.annotation.CacheEvict;
008import org.springframework.stereotype.Service;
009
010@Service
011public class CacheService {
012
013    private final Logger log = LoggerFactory.getLogger(getClass());
014
015    private CounterService counterService;
016
017    @Autowired
018    public CacheService(CounterService counterService) {
019        this.counterService = counterService;
020    }
021
022    @CacheEvict("translations")
023    public void evictTranslation(String source, String target, String text) {
024        counterService.increment("services.cache.key.eviction");
025        log.info("Key w/ source [{}], target [{}], and text [{}] removed from cache!", source, target, text);
026    }
027
028}
