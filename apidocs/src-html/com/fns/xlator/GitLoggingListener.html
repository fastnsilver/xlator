020package com.fns.xlator;
021
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024import org.springframework.beans.factory.NoSuchBeanDefinitionException;
025import org.springframework.boot.context.event.ApplicationStartedEvent;
026import org.springframework.boot.info.GitProperties;
027import org.springframework.context.ApplicationListener;
028import org.springframework.stereotype.Component;
029
030@Component
031// @see https://github.com/spring-projects/spring-boot/issues/11950
032public class GitLoggingListener implements ApplicationListener<ApplicationStartedEvent> {
033        
034        private final static Logger LOGGER = LoggerFactory.getLogger(GitLoggingListener.class);
035
036        @Override
037        public void onApplicationEvent(final ApplicationStartedEvent event) {
038                try {
039                        GitProperties git = event.getApplicationContext().getBean(GitProperties.class);
040                        LOGGER.info("This application was built from git commit id: {}", git.getShortCommitId());
041                } catch (NoSuchBeanDefinitionException e) {
042                        LOGGER.info("This application was not built in a git repository.");
043                }
044        }
045}
