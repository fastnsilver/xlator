020package com.fns.xlator.controller;
021
022import java.util.ArrayList;
023import java.util.Arrays;
024import java.util.List;
025
026import javax.servlet.http.HttpServletRequest;
027
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030import org.springframework.beans.factory.annotation.Autowired;
031import org.springframework.beans.factory.annotation.Value;
032import org.springframework.http.ResponseEntity;
033import org.springframework.util.Assert;
034import org.springframework.web.bind.annotation.DeleteMapping;
035import org.springframework.web.bind.annotation.ExceptionHandler;
036import org.springframework.web.bind.annotation.GetMapping;
037import org.springframework.web.bind.annotation.PathVariable;
038import org.springframework.web.bind.annotation.PostMapping;
039import org.springframework.web.bind.annotation.RequestBody;
040import org.springframework.web.bind.annotation.RequestMapping;
041import org.springframework.web.bind.annotation.RestController;
042
043import com.fns.xlator.client.api.CacheService;
044import com.fns.xlator.client.api.TranslationService;
045import com.fns.xlator.client.impl.TranslationException;
046import com.fns.xlator.model.ErrorResponse;
047import com.fns.xlator.model.Translation;
048import com.fns.xlator.model.TranslationRequest;
049import com.fns.xlator.model.Translations;
050
051import io.micrometer.core.annotation.Timed;
052
053
054@RestController
055@RequestMapping("/translation")
056public class TranslationController {
057
058    private final Logger log = LoggerFactory.getLogger(getClass());
059
060    private CacheService cacheService;
061    private TranslationService translationService;
062    private String defaultLocale;
063    private int maximumTranslations;
064
065    @Autowired
066    public TranslationController(CacheService cacheService, TranslationService translationService,
067            @Value("${app.defaults.locale}") String defaultLocale,
068            @Value("${app.limits.translationsPerRequest}") int maximumTranslations) {
069        this.cacheService = cacheService;
070        this.translationService = translationService;
071        this.defaultLocale = defaultLocale;
072        this.maximumTranslations = maximumTranslations;
073    }
074
075    
076    protected ResponseEntity<Translations> fetchTranslations(TranslationRequest[] translationRequests,
077            HttpServletRequest request) {
078        List<Translation> translations = new ArrayList<>();
079        List<ErrorResponse> errors = new ArrayList<>();
080        Assert.isTrue(translationRequests.length <= maximumTranslations, String.format(
081                "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
082        for (TranslationRequest tr : translationRequests) {
083            try {
084                translations.add(translationService.obtainTranslation(tr.getSource(), tr.getTarget(), tr.getText()));
085            } catch (IllegalArgumentException | TranslationException e) {
086                errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
087                        e.getMessage()));
088            }
089        }
090        Translations result = new Translations(translations, errors);
091        return ResponseEntity.ok(result);
092    }
093
094    protected ResponseEntity<Translations> fetchTranslations(String source, String targets, String text,
095            HttpServletRequest request) {
096        List<Translation> translations = new ArrayList<>();
097        List<ErrorResponse> errors = new ArrayList<>();
098        List<String> t = Arrays.asList(targets.split("\\s*,\\s*"));
099        Assert.isTrue(t.size() <= maximumTranslations, String.format(
100                "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
101        for (String target : t) {
102            try {
103                translations.add(translationService.obtainTranslation(source, target, text));
104            } catch (IllegalArgumentException | TranslationException e) {
105                errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
106                        e.getMessage()));
107            }
108        }
109        Translations result = new Translations(translations, errors);
110        return ResponseEntity.ok(result);
111    }
112
113    
114    @Timed
115    @GetMapping(value = "/source/{src}/target/{target}/text/{text}")
116    public ResponseEntity<Translation> obtainTranslation(@PathVariable("src") String source,
117            @PathVariable("target") String target, @PathVariable("text") String text) {
118        return ResponseEntity.ok(translationService.obtainTranslation(source, target, text));
119    }
120
121    @Timed
122    // convenience, uses a configurable default locale
123    @GetMapping(value = "/target/{target}/text/{text}")
124    public ResponseEntity<Translation> obtainTranslation(@PathVariable("target") String target,
125            @PathVariable("text") String text) {
126        return ResponseEntity.ok(translationService.obtainTranslation(defaultLocale, target, text));
127    }
128
129    @Timed
130    // convenience, source x n target combinations for a text
131    @GetMapping(value = "/source/{src}/targets/{targets}/text/{text}")
132    public ResponseEntity<Translations> obtainTranslations(@PathVariable("src") String source,
133            @PathVariable("targets") String targets, @PathVariable("text") String text, HttpServletRequest request) {
134        return fetchTranslations(source, targets, text, request);
135    }
136
137    @Timed
138    // convenience, default configurable locale x n target combinations for a text
139    @GetMapping(value = "/targets/{targets}/text/{text}")
140    public ResponseEntity<Translations> obtainTranslations(@PathVariable("targets") String targets,
141            @PathVariable("text") String text, HttpServletRequest request) {
142        return fetchTranslations(defaultLocale, targets, text, request);
143    }
144
145    @Timed
146    // most flexible option
147    @PostMapping(value = "/")
148    public ResponseEntity<Translations> obtainTranslations(@RequestBody TranslationRequest[] translationRequests,
149            HttpServletRequest request) {
150        return fetchTranslations(translationRequests, request);
151    }
152
153    @Timed
154    @DeleteMapping(value = "/source/{src}/target/{target}/text/{text}")
155    public ResponseEntity<?> invalidateCacheKey(@PathVariable("src") String source,
156            @PathVariable("target") String target, @PathVariable("text") String text) {
157        cacheService.evictTranslation(source, target, text);
158        log.info(String.format("Evicted cache key for /source/%s/target/%s/text/%s", source, target, text));
159        return ResponseEntity.noContent().build();
160    }
161
162    
163    @ExceptionHandler({ IllegalArgumentException.class, TranslationException.class })
164    protected ResponseEntity<ErrorResponse> badRequest(Exception e, HttpServletRequest hsr) {
165        ErrorResponse er = new ErrorResponse(hsr.getMethod(), hsr.getRequestURI(), hsr.getQueryString(),
166                e.getMessage());
167        log.error(er.asLog());
168        return ResponseEntity.badRequest().body(er);
169    }

171}
