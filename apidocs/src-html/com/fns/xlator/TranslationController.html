001package com.fns.xlator;
002
003import com.fns.xlator.client.api.TranslationService;
004import com.fns.xlator.model.ErrorResponse;
005import com.fns.xlator.model.Translation;
006import com.fns.xlator.model.TranslationRequest;
007import com.fns.xlator.model.Translations;
008
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011import org.springframework.beans.factory.annotation.Autowired;
012import org.springframework.beans.factory.annotation.Value;
013import org.springframework.http.ResponseEntity;
014import org.springframework.util.Assert;
015import org.springframework.web.bind.annotation.ExceptionHandler;
016import org.springframework.web.bind.annotation.PathVariable;
017import org.springframework.web.bind.annotation.RequestBody;
018import org.springframework.web.bind.annotation.RequestMapping;
019import org.springframework.web.bind.annotation.RequestMethod;
020import org.springframework.web.bind.annotation.RestController;
021
022import java.util.ArrayList;
023import java.util.Arrays;
024import java.util.List;
025
026import javax.servlet.http.HttpServletRequest;
027
028@RestController
029@RequestMapping("/translation")
030public class TranslationController {
031
032    private final Logger log = LoggerFactory.getLogger(getClass());
033    private TranslationService translationService;
034    private String defaultLocale;
035    private int maximumTranslations;
036
037    @Autowired
038    public TranslationController(TranslationService translationService,
039            @Value("${app.defaults.locale}") String defaultLocale,
040            @Value("${app.limits.translationsPerRequest}") int maximumTranslations) {
041        this.translationService = translationService;
042        this.defaultLocale = defaultLocale;
043    }
044
045
046    @RequestMapping(value = "/source/{src}/target/{target}/text/{text}", method = RequestMethod.GET)
047    public ResponseEntity<Translation> obtainTranslation(@PathVariable("src") String source,
048            @PathVariable("target") String target, @PathVariable("text") String text) {
049        return ResponseEntity.ok(translationService.obtainTranslation(source, target, text));
050    }
051
052    // convenience, uses a configurable default locale
053    @RequestMapping(value = "/target/{target}/text/{text}", method = RequestMethod.GET)
054    public ResponseEntity<Translation> obtainTranslation(@PathVariable("target") String target,
055            @PathVariable("text") String text) {
056        return ResponseEntity.ok(translationService.obtainTranslation(defaultLocale, target, text));
057    }
058
059    // convenience, source x n target combinations for a text
060    @RequestMapping(value = "/source/{src}/targets/{targets}/text/{text}", method = RequestMethod.GET)
061    public ResponseEntity<Translations> obtainTranslations(@PathVariable("src") String source,
062            @PathVariable("targets") String targets, @PathVariable("text") String text, HttpServletRequest request) {
063        return fetchTranslations(source, targets, text, request);
064    }
065
066    // convenience, default configurable locale x n target combinations for a text
067    @RequestMapping(value = "/targets/{targets}/text/{text}", method = RequestMethod.GET)
068    public ResponseEntity<Translations> obtainTranslations(@PathVariable("targets") String targets,
069            @PathVariable("text") String text, HttpServletRequest request) {
070        return fetchTranslations(defaultLocale, targets, text, request);
071    }
072
073    // most flexible option
074    @RequestMapping(value = "/", method = RequestMethod.POST)
075    public ResponseEntity<Translations> obtainTranslations(@RequestBody TranslationRequest[] translationRequests,
076            HttpServletRequest request) {
077        return fetchTranslations(translationRequests, request);
078    }
079
080    protected ResponseEntity<Translations> fetchTranslations(TranslationRequest[] translationRequests,
081            HttpServletRequest request) {
082        List<Translation> translations = new ArrayList<>();
083        List<ErrorResponse> errors = new ArrayList<>();
084        Assert.isTrue(translationRequests.length <= maximumTranslations, String.format(
085                "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
086        for (TranslationRequest tr : translationRequests) {
087            try {
088                translations.add(translationService.obtainTranslation(tr.getSource(), tr.getTarget(), tr.getText()));
089            } catch (IllegalArgumentException iae) {
090                errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
091                        iae.getMessage()));
092            }
093        }
094        Translations result = new Translations(translations, errors);
095        return ResponseEntity.ok(result);
096    }
097
098    protected ResponseEntity<Translations> fetchTranslations(String source, String targets, String text,
099            HttpServletRequest request) {
100        List<Translation> translations = new ArrayList<>();
101        List<ErrorResponse> errors = new ArrayList<>();
102        List<String> t = Arrays.asList(targets.split("\\s*,\\s*"));
103        Assert.isTrue(t.size() <= maximumTranslations, String.format(
104                "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
105        for (String target : t) {
106            try {
107                translations.add(translationService.obtainTranslation(source, target, text));
108            } catch (IllegalArgumentException iae) {
109                errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
110                        iae.getMessage()));
111            }
112        }
113        Translations result = new Translations(translations, errors);
114        return ResponseEntity.ok(result);
115    }
116
117    @RequestMapping(value = "/source/{src}/target/{target}/text/{text}", method = RequestMethod.DELETE)
118    public ResponseEntity<?> invalidateCacheKey(@PathVariable("src") String source,
119            @PathVariable("target") String target, @PathVariable("text") String text) {
120        translationService.evictTranslation(source, target, text);
121        log.info(String.format("Evicted cache key for /source/%s/target/%s/text/%s", source, target, text));
122        return ResponseEntity.noContent().build();
123    }
124
125    @ExceptionHandler({ IllegalArgumentException.class })
126    protected ResponseEntity<ErrorResponse> badRequest(Exception e, HttpServletRequest hsr) {
127        ErrorResponse er = new ErrorResponse(hsr.getMethod(), hsr.getRequestURI(), hsr.getQueryString(),
128                e.getMessage());
129        log.error(er.asLog());
130        return ResponseEntity.badRequest().body(er);
131    }
132
133}
