001package com.fns.xlator;
002
003import com.fns.xlator.client.api.CacheService;
004import com.fns.xlator.client.api.TranslationService;
005import com.fns.xlator.model.ErrorResponse;
006import com.fns.xlator.model.Translation;
007import com.fns.xlator.model.TranslationRequest;
008import com.fns.xlator.model.Translations;
009
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012import org.springframework.beans.factory.annotation.Autowired;
013import org.springframework.beans.factory.annotation.Value;
014import org.springframework.http.ResponseEntity;
015import org.springframework.util.Assert;
016import org.springframework.web.bind.annotation.ExceptionHandler;
017import org.springframework.web.bind.annotation.PathVariable;
018import org.springframework.web.bind.annotation.RequestBody;
019import org.springframework.web.bind.annotation.RequestMapping;
020import org.springframework.web.bind.annotation.RequestMethod;
021import org.springframework.web.bind.annotation.RestController;
022
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026
027import javax.servlet.http.HttpServletRequest;
028
029import io.swagger.annotations.Api;
030import io.swagger.annotations.ApiOperation;
031import io.swagger.annotations.ApiResponse;
032import io.swagger.annotations.ApiResponses;
033
034@RestController
035@RequestMapping("/translation")
036@Api(value = "translation", produces = "application/json")
037public class TranslationController {
038
039    private final Logger log = LoggerFactory.getLogger(getClass());
040
041    private CacheService cacheService;
042    private TranslationService translationService;
043    private String defaultLocale;
044    private int maximumTranslations;
045
046    @Autowired
047    public TranslationController(CacheService cacheService, TranslationService translationService,
048            @Value("${app.defaults.locale}") String defaultLocale,
049            @Value("${app.limits.translationsPerRequest}") int maximumTranslations) {
050        this.translationService = translationService;
051        this.defaultLocale = defaultLocale;
052    }
053
054    
055    protected ResponseEntity<Translations> fetchTranslations(TranslationRequest[] translationRequests,
056            HttpServletRequest request) {
057        List<Translation> translations = new ArrayList<>();
058        List<ErrorResponse> errors = new ArrayList<>();
059        Assert.isTrue(translationRequests.length <= maximumTranslations, String.format(
060                "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
061        for (TranslationRequest tr : translationRequests) {
062            try {
063                translations.add(translationService.obtainTranslation(tr.getSource(), tr.getTarget(), tr.getText()));
064            } catch (IllegalArgumentException iae) {
065                errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
066                        iae.getMessage()));
067            }
068        }
069        Translations result = new Translations(translations, errors);
070        return ResponseEntity.ok(result);
071    }
072
073    protected ResponseEntity<Translations> fetchTranslations(String source, String targets, String text,
074            HttpServletRequest request) {
075        List<Translation> translations = new ArrayList<>();
076        List<ErrorResponse> errors = new ArrayList<>();
077        List<String> t = Arrays.asList(targets.split("\\s*,\\s*"));
078        Assert.isTrue(t.size() <= maximumTranslations, String.format(
079                "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
080        for (String target : t) {
081            try {
082                translations.add(translationService.obtainTranslation(source, target, text));
083            } catch (IllegalArgumentException iae) {
084                errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
085                        iae.getMessage()));
086            }
087        }
088        Translations result = new Translations(translations, errors);
089        return ResponseEntity.ok(result);
090    }
091
092
093    @RequestMapping(value = "/source/{src}/target/{target}/text/{text}", method = RequestMethod.GET)
094    @ApiOperation(httpMethod = "GET", notes = "Obtain a translation.", value = "/translation/source/{src}/target/{target}/text/{text}")
095    @ApiResponses(value = { 
096            @ApiResponse(code = 200, message = "Successfully obtained translation."),
097            @ApiResponse(code = 400, message = "Bad request."),
098            @ApiResponse(code = 500, message = "Internal server error.") 
099    })
100    public ResponseEntity<Translation> obtainTranslation(@PathVariable("src") String source,
101            @PathVariable("target") String target, @PathVariable("text") String text) {
102        return ResponseEntity.ok(translationService.obtainTranslation(source, target, text));
103    }
104
105    // convenience, uses a configurable default locale
106    @RequestMapping(value = "/target/{target}/text/{text}", method = RequestMethod.GET)
107    @ApiOperation(httpMethod = "GET", notes = "Obtain a translation using default locale.", value = "/translation/target/{target}/text/{text}")
108    @ApiResponses(value = { 
109            @ApiResponse(code = 200, message = "Successfully obtained translation."),
110            @ApiResponse(code = 400, message = "Bad request."),
111            @ApiResponse(code = 500, message = "Internal server error.") 
112    })
113    public ResponseEntity<Translation> obtainTranslation(@PathVariable("target") String target,
114            @PathVariable("text") String text) {
115        return ResponseEntity.ok(translationService.obtainTranslation(defaultLocale, target, text));
116    }
117
118    // convenience, source x n target combinations for a text
119    @RequestMapping(value = "/source/{src}/targets/{targets}/text/{text}", method = RequestMethod.GET)
120    @ApiOperation(httpMethod = "GET", notes = "Obtain translations for each target locale.", value = "/translation/source/{src}/targets/{targets}/text/{text}")
121    @ApiResponses(value = { 
122            @ApiResponse(code = 200, message = "Obtained translations.  (Report errors with any target)."),
123            @ApiResponse(code = 400, message = "Bad request."),
124            @ApiResponse(code = 500, message = "Internal server error.") 
125    })
126    public ResponseEntity<Translations> obtainTranslations(@PathVariable("src") String source,
127            @PathVariable("targets") String targets, @PathVariable("text") String text, HttpServletRequest request) {
128        return fetchTranslations(source, targets, text, request);
129    }
130
131    // convenience, default configurable locale x n target combinations for a text
132    @RequestMapping(value = "/targets/{targets}/text/{text}", method = RequestMethod.GET)
133    @ApiOperation(httpMethod = "GET", notes = "Obtain translations for each target locale using default locale.", value = "/translation/targets/{targets}/text/{text}")
134    @ApiResponses(value = { 
135            @ApiResponse(code = 200, message = "Obtained translations.  (Report errors with any target)."),
136            @ApiResponse(code = 400, message = "Bad request."),
137            @ApiResponse(code = 500, message = "Internal server error.") 
138    })
139    public ResponseEntity<Translations> obtainTranslations(@PathVariable("targets") String targets,
140            @PathVariable("text") String text, HttpServletRequest request) {
141        return fetchTranslations(defaultLocale, targets, text, request);
142    }
143
144    // most flexible option
145    @RequestMapping(value = "/", method = RequestMethod.POST)
146    @ApiOperation(httpMethod = "POST", notes = "Obtain translations.", value = "/translation/")
147    @ApiResponses(value = { 
148            @ApiResponse(code = 200, message = "Obtained translations.  (Report errors with any source-target-text combination)."),
149            @ApiResponse(code = 400, message = "Bad request."),
150            @ApiResponse(code = 500, message = "Internal server error.") 
151    })
152    public ResponseEntity<Translations> obtainTranslations(@RequestBody TranslationRequest[] translationRequests,
153            HttpServletRequest request) {
154        return fetchTranslations(translationRequests, request);
155    }
156
157    @RequestMapping(value = "/source/{src}/target/{target}/text/{text}", method = RequestMethod.DELETE)
158    @ApiOperation(httpMethod = "DELETE", notes = "Evict cache-key for a previously obtained translation.", value = "/translation/source/{src}/target/{target}/text/{text}")
159    @ApiResponses(value = { 
160            @ApiResponse(code = 204, message = "Successfully evicted key."),
161            @ApiResponse(code = 500, message = "Internal server error.") 
162    })
163    public ResponseEntity<?> invalidateCacheKey(@PathVariable("src") String source,
164            @PathVariable("target") String target, @PathVariable("text") String text) {
165        cacheService.evictTranslation(source, target, text);
166        log.info(String.format("Evicted cache key for /source/%s/target/%s/text/%s", source, target, text));
167        return ResponseEntity.noContent().build();
168    }
169
170    
171    @ExceptionHandler({ IllegalArgumentException.class })
172    protected ResponseEntity<ErrorResponse> badRequest(Exception e, HttpServletRequest hsr) {
173        ErrorResponse er = new ErrorResponse(hsr.getMethod(), hsr.getRequestURI(), hsr.getQueryString(),
174                e.getMessage());
175        log.error(er.asLog());
176        return ResponseEntity.badRequest().body(er);
177    }
