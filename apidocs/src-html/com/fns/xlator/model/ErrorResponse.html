001package com.fns.xlator.model;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnore;
005import com.fasterxml.jackson.annotation.JsonInclude;
006import com.fasterxml.jackson.annotation.JsonInclude.Include;
007import com.fasterxml.jackson.annotation.JsonProperty;
008import com.fasterxml.jackson.annotation.JsonPropertyOrder;
009
010import org.apache.commons.lang3.builder.ToStringBuilder;
011
012@JsonPropertyOrder(value = { "error", "method", "uri", "query" })
013@JsonInclude(Include.NON_EMPTY)
014public class ErrorResponse {
015    private String method;
016    private String uri;
017    private String query;
018    private String error;
019
020    @JsonCreator
021    public ErrorResponse(@JsonProperty String method, @JsonProperty String uri, @JsonProperty String query,
022            @JsonProperty String error) {
023        this.method = method;
024        this.uri = uri;
025        this.query = query;
026        this.error = error;
027    }
028
029    public String getMethod() {
030        return method;
031    }
032
033    public String getUri() {
034        return uri;
035    }
036
037    public String getQuery() {
038        return query;
039    }
040
041    public String getError() {
042        return error;
043    }
044
045    @JsonIgnore
046    public String asLog() {
047        return ToStringBuilder.reflectionToString(this);
048    }
049
050}
