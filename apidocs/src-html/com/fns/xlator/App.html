001package com.fns.xlator;
002
003import com.fns.xlator.client.api.TranslationService;
004import com.fns.xlator.client.impl.frengly.FrenglyClientSettings;
005import com.fns.xlator.client.impl.frengly.FrenglyTranslationService;
006import com.fns.xlator.client.impl.google.GoogleClientSettings;
007import com.fns.xlator.client.impl.google.GoogleTranslationService;
008
009import org.springframework.beans.factory.annotation.Autowired;
010import org.springframework.boot.SpringApplication;
011import org.springframework.boot.actuate.metrics.CounterService;
012import org.springframework.boot.autoconfigure.AutoConfigureAfter;
013import org.springframework.boot.autoconfigure.SpringBootApplication;
014import org.springframework.boot.autoconfigure.cache.CacheProperties;
015import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
016import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
017import org.springframework.boot.autoconfigure.data.redis.RedisAutoConfiguration;
018import org.springframework.cache.Cache;
019import org.springframework.cache.CacheManager;
020import org.springframework.cache.concurrent.ConcurrentMapCache;
021import org.springframework.cache.support.SimpleCacheManager;
022import org.springframework.cloud.aws.autoconfigure.cache.ElastiCacheAutoConfiguration;
023import org.springframework.cloud.aws.cache.config.annotation.CacheClusterConfig;
024import org.springframework.cloud.aws.cache.config.annotation.EnableElastiCache;
025import org.springframework.context.annotation.Bean;
026import org.springframework.context.annotation.Configuration;
027import org.springframework.context.annotation.Profile;
028import org.springframework.data.redis.cache.RedisCacheManager;
029import org.springframework.data.redis.core.RedisTemplate;
030import org.springframework.web.client.RestTemplate;
031
032import java.util.ArrayList;
033import java.util.List;
034
035// special thanks to https://github.com/spring-cloud-samples/aws-refapp/blob/master/src/main/java/org/springframework/cloud/aws/sample/ReferenceApplication.java
036
037@SpringBootApplication(exclude = ElastiCacheAutoConfiguration.class)
038public class App {
039
040    @Configuration
041    @ConditionalOnProperty(prefix = "app.defaults", name = "service", havingValue = "google")
042    static class GoogleApiConfig {
043
044        @Bean
045        public GoogleClientSettings googleClientSettings() {
046            return new GoogleClientSettings();
047        }
048
049        @Bean
050        public TranslationService googleTranslationService(RestTemplate restTemplate, GoogleClientSettings settings,
051                CounterService counterService) {
052            return new GoogleTranslationService(restTemplate, settings, counterService);
053        }
054
055    }
056
057    @Configuration
058    @ConditionalOnProperty(prefix = "app.defaults", name = "service", havingValue = "frengly")
059    static class FrenglyApiConfig {
060
061        @Bean
062        public FrenglyClientSettings frenglyClientSettings() {
063            return new FrenglyClientSettings();
064        }
065
066        @Bean
067        public TranslationService frenglyTranslationService(RestTemplate restTemplate, FrenglyClientSettings settings,
068                CounterService counterService) {
069            return new FrenglyTranslationService(restTemplate, settings, counterService);
070        }
071    }
072
073    @Profile(value = { "!aws", "!docker" })
074    @Configuration
075    static class LocalCacheConfig {
076
077        @Bean
078        public CacheManager createSimpleCacheManager() {
079            SimpleCacheManager simpleCacheManager = new SimpleCacheManager();
080            List<Cache> caches = new ArrayList<>();
081            caches.add(new ConcurrentMapCache("translations"));
082            simpleCacheManager.setCaches(caches);
083            return simpleCacheManager;
084        }
085
086    }
087
088    @Profile("aws")
089    @Configuration
090    @EnableElastiCache({ @CacheClusterConfig(name = "translations" /*, expiration = 1209600 */) })
091    static class ElastiCacheConfig {
092
093    }
094
095    @Profile("docker")
096    @Configuration
097    @AutoConfigureAfter(RedisAutoConfiguration.class)
098    @ConditionalOnBean(RedisTemplate.class)
099    static class RedisCacheConfig {
100
101        @Autowired
102        private CacheProperties cacheProperties;
103
104        @Bean
105        public RedisCacheManager cacheManager(RedisTemplate<Object, Object> redisTemplate) {
106            RedisCacheManager cacheManager = new RedisCacheManager(redisTemplate);
107            List<String> cacheNames = this.cacheProperties.getCacheNames();
108            if (!cacheNames.isEmpty()) {
109                cacheManager.setCacheNames(cacheNames);
110            }
111            return cacheManager;
112        }
113    }
114
115
116    public static void main(String[] args) {
117        SpringApplication.run(App.class, args);
118    }
119}
