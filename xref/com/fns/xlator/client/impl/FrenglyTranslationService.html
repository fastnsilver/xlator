23  @Service

32      @Autowired
33      public FrenglyTranslationService(RestTemplate restTemplate, FrenglyClientSettings settings,
34              CounterService counterService) {
35          this.restTemplate = restTemplate;
36          this.settings = settings;
37          this.counterService = counterService;
38      }

40      private Locale parseLocale(String locale) {
41          String[] parts = locale.split("_");
42          switch (parts.length) {
43              case 3:
44                  return new Locale(parts[0], parts[1], parts[2]);
45              case 2:
46                  return new Locale(parts[0], parts[1]);
47              case 1:
48                  return new Locale(parts[0]);
49              default:
50                  throw new IllegalArgumentException("Invalid locale: " + locale);
51          }
52      }
53  
54      private boolean isValidLocale(String code) {
55          try {
56              Locale locale = parseLocale(code);
57              return Arrays.asList(Locale.getAvailableLocales()).contains(locale);
58          } catch (MissingResourceException e) {
59              return false;
60          }
61      }
62  
63      protected void validateTranslationArguments(String source, String target, String text) {
64          Assert.hasText(target, "[Assertion failed] - Target locale code must not be null, empty or blank!");
65          Assert.hasText(text, "[Assertion failed] - Text to translate must not be null, empty or blank!");
66          Assert.isTrue(isValidLocale(source), String.format("[Assertion failed] - Source [%s] is not a valid Locale!", source));
67          Assert.isTrue(isValidLocale(target), String.format("[Assertion failed] - Target [%s] is not a valid Locale!", target));
68      }
69  
70      protected FrenglyTranslation tryRequest(String source, String target, String text, int retries) {
71          URI uri = UriComponentsBuilder
72                  .newInstance()
73                  .scheme("http")
74                  .host(settings.getHost())
75                  .port(80)
76                  .queryParam("src", source)
77                  .queryParam("dest", target)
78                  .queryParam("text", text)
79                  .queryParam("email", settings.getEmail())
80                  .queryParam("password", settings.getPassword())
81                  .queryParam("outformat", "json")
82                  .build()
83                  .encode()
84                  .toUri();
85          log.info(String.format("Obtaining translation from %s", uri.toASCIIString()));
86  
87          FrenglyTranslation result = null;
88          for (int i = 0; i < retries; i++) {
89              try {
90                  result = restTemplate.getForObject(uri, FrenglyTranslation.class);
91                  break;
92              } catch (RestClientException e) {
93                  try {
94                      Thread.sleep((i + 1) * 500);
95                  } catch (InterruptedException ie) {
96                      // do nothing
97                  }
98              }
99              if (i == retries - 1) {
100                 result = new FrenglyTranslation(source, target, text, "Could not obtain translation!"); // FIXME i18n response
101             }
102         }
103         return result;
104     }
105 
106     @Override
107     @Cacheable(cacheNames = "translations")
108     public Translation obtainTranslation(String src, String target, String text) {
109         counterService.increment("services.frengly.translation.invoked");
110         String source = src;
111         if (source == null) {
112             source = settings.getDefaultSource();
113         }
114         validateTranslationArguments(source, target, text);
115         if (src.equals(target)) {
116             return new Translation(src, target, text, text);
117         }
118         FrenglyTranslation ft = tryRequest(source, target, text, settings.getRetries());
119         return new Translation(ft.getSrc(), ft.getDest(), ft.getText(), ft.getTranslation());
120     }
121 
122     @Override
123     @CacheEvict("translations")
124     public void evictTranslation(String source, String target, String text) {
125         counterService.increment("services.frengly.cacheKey.eviction");
126         log.info("Key w/ source [{}], target [{}], and text [{}] removed from cache!");
127     }
