19  public class FrenglyTranslationService implements TranslationService {
20  
21      private final Logger log = LoggerFactory.getLogger(getClass());
22  
23      private RestTemplate restTemplate;
24      private FrenglyClientSettings settings;
25      private CounterService counterService;
26  
27      @Autowired
28      public FrenglyTranslationService(RestTemplate restTemplate, FrenglyClientSettings settings,
29              CounterService counterService) {
30          this.restTemplate = restTemplate;
31          this.settings = settings;
32          this.counterService = counterService;
33      }
34  
35      // @see http://www.frengly.com/#!/api
36      protected FrenglyTranslation tryRequest(String source, String target, String text, int retries) {
37          URI uri = UriComponentsBuilder
38                  .newInstance()
39                  .scheme("http")
40                  .host(settings.getHost())
41                  .port(80)
42                  .queryParam("src", source)
43                  .queryParam("dest", target)
44                  .queryParam("text", text)
45                  .queryParam("email", settings.getEmail())
46                  .queryParam("password", settings.getPassword())
47                  .queryParam("outformat", "json")
48                  .build()
49                  .encode().toUri();
50          log.info(String.format("Obtaining translation from %s", uri.toASCIIString()));
51  
52          FrenglyTranslation result = null;
53          for (int i = 0; i < retries; i++) {
54              try {
55                  result = restTemplate.getForObject(uri, FrenglyTranslation.class);
56                  break;
57              } catch (RestClientException e) {
58                  try {
59                      Thread.sleep((i + 1) * 500);
60                  } catch (InterruptedException ie) {
61                      // do nothing
62                  }
63              }
64              if (i == retries - 1) {
65                  throw new TranslationException(source, target, text);
66              }
67          }
68          return result;
69      }
70  
71      @Override
72      @Cacheable(cacheNames = "translations")
73      public Translation obtainTranslation(String src, String target, String text) {
74          counterService.increment("services.frengly.translation.invoked");
75          String source = src;
76          if (source == null) {
77              source = settings.getDefaultSource();
78          }
79          ValidationUtils.validateTranslationArguments(source, target, text);
80          if (src.equals(target)) {
81              return new Translation(src, target, text, text);
82          }
83          FrenglyTranslation ft = tryRequest(source, target, text, settings.getRetries());
84          return new Translation(ft.getSrc(), ft.getDest(), ft.getText(), ft.getTranslation());
85      }
