13      private static Locale parseLocale(String locale) {
14          String[] parts = locale.split("_");
15          switch (parts.length) {
16              case 3:
17                  return new Locale(parts[0], parts[1], parts[2]);
18              case 2:
19                  return new Locale(parts[0], parts[1]);
20              case 1:
21                  return new Locale(parts[0]);
22              default:
23                  throw new IllegalArgumentException("Invalid locale: " + locale);
24          }
25      }
26  
27      private static boolean isValidLocale(String code) {
28          try {
29              Locale locale = parseLocale(code);
30              return Arrays.asList(Locale.getAvailableLocales()).contains(locale);
31          } catch (MissingResourceException e) {
32              return false;
33          }
34      }
35  
36      public static void validateTranslationArguments(String source, String target, String text) {
37          Assert.hasText(target, "[Assertion failed] - Target locale code must not be null, empty or blank!");
38          Assert.hasText(text, "[Assertion failed] - Text to translate must not be null, empty or blank!");
39          Assert.isTrue(isValidLocale(source), String.format("[Assertion failed] - Source [%s] is not a valid Locale!", source));
40          Assert.isTrue(isValidLocale(target), String.format("[Assertion failed] - Target [%s] is not a valid Locale!", target));
41      }
