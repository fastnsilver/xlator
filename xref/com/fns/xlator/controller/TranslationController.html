24  import java.util.List;

26  import javax.servlet.http.HttpServletRequest;

28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;

32  import org.springframework.http.ResponseEntity;

35  import org.springframework.web.bind.annotation.ExceptionHandler;

40  import org.springframework.web.bind.annotation.RequestMapping;
41  import org.springframework.web.bind.annotation.RestController;

43  import com.fns.xlator.client.api.CacheService;
44  import com.fns.xlator.client.api.TranslationService;
45  import com.fns.xlator.client.impl.TranslationException;
46  import com.fns.xlator.model.ErrorResponse;
47  import com.fns.xlator.model.Translation;
48  import com.fns.xlator.model.TranslationRequest;
49  import com.fns.xlator.model.Translations;

51  import io.micrometer.core.annotation.Timed;
52  
53  
54  @RestController
55  @RequestMapping("/translation")
56  public class TranslationController {
57  
58      private final Logger log = LoggerFactory.getLogger(getClass());
59  
60      private CacheService cacheService;
61      private TranslationService translationService;
62      private String defaultLocale;
63      private int maximumTranslations;
64  
65      @Autowired
66      public TranslationController(CacheService cacheService, TranslationService translationService,
67              @Value("${app.defaults.locale}") String defaultLocale,
68              @Value("${app.limits.translationsPerRequest}") int maximumTranslations) {
69          this.cacheService = cacheService;
70          this.translationService = translationService;
71          this.defaultLocale = defaultLocale;
72          this.maximumTranslations = maximumTranslations;
73      }
74  
75      
76      protected ResponseEntity<Translations> fetchTranslations(TranslationRequest[] translationRequests,
77              HttpServletRequest request) {
78          List<Translation> translations = new ArrayList<>();
79          List<ErrorResponse> errors = new ArrayList<>();
80          Assert.isTrue(translationRequests.length <= maximumTranslations, String.format(
81                  "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
82          for (TranslationRequest tr : translationRequests) {
83              try {
84                  translations.add(translationService.obtainTranslation(tr.getSource(), tr.getTarget(), tr.getText()));
85              } catch (IllegalArgumentException | TranslationException e) {
86                  errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
87                          e.getMessage()));
88              }
89          }
90          Translations result = new Translations(translations, errors);
91          return ResponseEntity.ok(result);
92      }
93  
94      protected ResponseEntity<Translations> fetchTranslations(String source, String targets, String text,
95              HttpServletRequest request) {
96          List<Translation> translations = new ArrayList<>();
97          List<ErrorResponse> errors = new ArrayList<>();
98          List<String> t = Arrays.asList(targets.split("\\s*,\\s*"));
99          Assert.isTrue(t.size() <= maximumTranslations, String.format(
100                 "[Assertion failed] - No more than %d translations may be requested at once!", maximumTranslations));
101         for (String target : t) {
102             try {
103                 translations.add(translationService.obtainTranslation(source, target, text));
104             } catch (IllegalArgumentException | TranslationException e) {
105                 errors.add(new ErrorResponse(request.getMethod(), request.getRequestURI(), request.getQueryString(),
106                         e.getMessage()));
107             }
108         }
109         Translations result = new Translations(translations, errors);
110         return ResponseEntity.ok(result);
111     }
112 
113     
114     @Timed
115     @GetMapping(value = "/source/{src}/target/{target}/text/{text}")
116     public ResponseEntity<Translation> obtainTranslation(@PathVariable("src") String source,
117             @PathVariable("target") String target, @PathVariable("text") String text) {
118         return ResponseEntity.ok(translationService.obtainTranslation(source, target, text));
119     }
120 
121     @Timed
122     // convenience, uses a configurable default locale
123     @GetMapping(value = "/target/{target}/text/{text}")
124     public ResponseEntity<Translation> obtainTranslation(@PathVariable("target") String target,
125             @PathVariable("text") String text) {
126         return ResponseEntity.ok(translationService.obtainTranslation(defaultLocale, target, text));
127     }
128 
129     @Timed
130     // convenience, source x n target combinations for a text
131     @GetMapping(value = "/source/{src}/targets/{targets}/text/{text}")
132     public ResponseEntity<Translations> obtainTranslations(@PathVariable("src") String source,
133             @PathVariable("targets") String targets, @PathVariable("text") String text, HttpServletRequest request) {
134         return fetchTranslations(source, targets, text, request);
135     }
136 
137     @Timed
138     // convenience, default configurable locale x n target combinations for a text
139     @GetMapping(value = "/targets/{targets}/text/{text}")
140     public ResponseEntity<Translations> obtainTranslations(@PathVariable("targets") String targets,
141             @PathVariable("text") String text, HttpServletRequest request) {
142         return fetchTranslations(defaultLocale, targets, text, request);
143     }
144 
145     @Timed
146     // most flexible option
147     @PostMapping(value = "/")
148     public ResponseEntity<Translations> obtainTranslations(@RequestBody TranslationRequest[] translationRequests,
149             HttpServletRequest request) {
150         return fetchTranslations(translationRequests, request);
151     }
152 
153     @Timed
154     @DeleteMapping(value = "/source/{src}/target/{target}/text/{text}")
155     public ResponseEntity<?> invalidateCacheKey(@PathVariable("src") String source,
156             @PathVariable("target") String target, @PathVariable("text") String text) {
157         cacheService.evictTranslation(source, target, text);
158         log.info(String.format("Evicted cache key for /source/%s/target/%s/text/%s", source, target, text));
159         return ResponseEntity.noContent().build();
160     }
161 
162     
163     @ExceptionHandler({ IllegalArgumentException.class, TranslationException.class })
164     protected ResponseEntity<ErrorResponse> badRequest(Exception e, HttpServletRequest hsr) {
165         ErrorResponse er = new ErrorResponse(hsr.getMethod(), hsr.getRequestURI(), hsr.getQueryString(),
166                 e.getMessage());
167         log.error(er.asLog());
168         return ResponseEntity.badRequest().body(er);
169     }
