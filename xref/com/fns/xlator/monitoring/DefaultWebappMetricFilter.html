16  package com.fns.xlator.monitoring;
17  
18  import java.util.HashMap;
19  import java.util.Map;
20  
21  import com.codahale.metrics.servlet.AbstractInstrumentedFilter;
22  
23  // like InstrumentedFilter except that it adds 2 more status codes: ACCEPTED (202) and CONFLICT (409)
24  class DefaultWebappMetricFilter extends AbstractInstrumentedFilter {
25      public static final String REGISTRY_ATTRIBUTE = DefaultWebappMetricFilter.class.getName() + ".registry";
26  
27      private static final String NAME_PREFIX = "responseCodes.";
28      private static final int OK = 200;
29      private static final int CREATED = 201;
30      private static final int ACCEPTED = 202;
31      private static final int NO_CONTENT = 204;
32      private static final int BAD_REQUEST = 400;
33      private static final int CONFLICT = 409;
34      private static final int NOT_FOUND = 404;
35      private static final int SERVER_ERROR = 500;
36  
37      /**
38       * Creates a new instance of the filter.
39       */
40      public DefaultWebappMetricFilter() {
41          super(REGISTRY_ATTRIBUTE, createMeterNamesByStatusCode(), NAME_PREFIX + "other");
42      }
43  
44      private static Map<Integer, String> createMeterNamesByStatusCode() {
45          final Map<Integer, String> meterNamesByStatusCode = new HashMap<Integer, String>(8);
46          meterNamesByStatusCode.put(OK, NAME_PREFIX + "ok");
47          meterNamesByStatusCode.put(CREATED, NAME_PREFIX + "created");
48          meterNamesByStatusCode.put(ACCEPTED, NAME_PREFIX + "accepted");
49          meterNamesByStatusCode.put(NO_CONTENT, NAME_PREFIX + "noContent");
50          meterNamesByStatusCode.put(BAD_REQUEST, NAME_PREFIX + "badRequest");
51          meterNamesByStatusCode.put(NOT_FOUND, NAME_PREFIX + "notFound");
52          meterNamesByStatusCode.put(CONFLICT, NAME_PREFIX + "conflict");
53          meterNamesByStatusCode.put(SERVER_ERROR, NAME_PREFIX + "serverError");
54          return meterNamesByStatusCode;
55      }
56  }
